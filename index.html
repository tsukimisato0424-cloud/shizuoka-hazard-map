// マップ初期化 const map = L.map('map', { zoomControl: false }).setView([34.9756, 138.3828], 8); // ズームボタンを右下に配置 L.control.zoom({ position: 'bottomright' }).addTo(map); // ベースマップ（国土地理院 淡色地図） L.tileLayer('https://cyberjapandata.gsi.go.jp/xyz/pale/{z}/{x}/{y}.png', { attribution: '© 国土地理院' }).addTo(map); // ハザードマップレイヤーの定義 const flood = L.tileLayer('https://disaportal.gsi.go.jp/raster/01_flood_l2_shinsuishin_data/{z}/{x}/{y}.png', { opacity: 0.6 }); const landslide = L.tileLayer('https://disaportal.gsi.go.jp/raster/05_dosekiryukeikaikuiki/{z}/{x}/{y}.png', { opacity: 0.7 }); const tsunami = L.tileLayer('https://disaportal.gsi.go.jp/raster/04_tsunami_newlegend_data/{z}/{x}/{y}.png', { opacity: 0.6 }); // ★ 変更点： `flood.addTo(map);` を削除しました。 // これにより、起動時は真っ白（通常の地図）になり、チェックを入れた時だけ重なるようになります。 // レイヤーコントロールを左上（検索窓の下）に配置し、最初から展開 L.control.layers(null, { "洪水（想定最大規模）": flood, "土砂災害警戒区域": landslide, "津波浸水想定": tsunami }, { position: "topleft", collapsed: false // 最初からメニューを開いた状態 }).addTo(map); let currentLocation = null, destination = null, routingControl = null, currentMarker = null, destMarker = null; // ルート描画関数 function drawRoute() { if (!currentLocation || !destination) return; if (routingControl) map.removeControl(routingControl); routingControl = L.Routing.control({ waypoints: [currentLocation, destination], routeWhileDragging: false, addWaypoints: false, draggableWaypoints: false, lineOptions: { styles: [{ color: '#1a73e8', opacity: 0.8, weight: 6 }] } }).addTo(map); } // 検索コントロール（最上段いっぱいに広げる） L.Control.geocoder({ defaultMarkGeocode: false, placeholder: " 🔍 場所、住所、施設名を入力...", collapsed: false }) .on("markgeocode", e => { destination = e.geocode.center; if (destMarker) map.removeLayer(destMarker); destMarker = L.marker(destination).addTo(map).bindPopup(e.geocode.name).openPopup(); map.setView(destination, 15); drawRoute(); }).addTo(map); // 現在地ボタンの処理 document.getElementById("locationBtn").onclick = () => { navigator.geolocation.getCurrentPosition(pos => { currentLocation = L.latLng(pos.coords.latitude, pos.coords.longitude); if (currentMarker) map.removeLayer(currentMarker); currentMarker = L.marker(currentLocation).addTo(map).bindPopup("現在地").openPopup(); map.setView(currentLocation, 15); drawRoute(); }, err => alert("現在地取得失敗: " + err.code), { enableHighAccuracy: true }); };
